import numpy as np

def compute_reward(agent_name, observation, global_state, actions, world):
    components = {}
    
    # Extract global state information
    agent_positions = global_state['agent_positions']
    agent_velocities = global_state['agent_velocities']
    prey_position = global_state['prey_position']
    prey_velocity = global_state['prey_velocity']
    distances_to_prey = global_state['distances_to_prey']
    inter_agent_distances = global_state['inter_agent_distances']
    is_adversary = global_state['is_adversary']
    adversary_indices = global_state['adversary_indices']
    prey_indices = global_state['prey_indices']
    world_size = global_state['world_size']
    capture_threshold = global_state['capture_threshold']
    
    # Find current agent index
    all_agent_names = [f"agent_{i}" for i in range(len(agent_positions))]
    agent_idx = all_agent_names.index(agent_name) if agent_name in all_agent_names else -1
    
    if is_adversary:
        # Pursuer reward components
        agent_idx_in_adversaries = adversary_indices.index(agent_idx)
        dist_to_prey = distances_to_prey[agent_idx_in_adversaries]
        
        # Distance reward (negative since we want to minimize distance)
        distance_reward = -0.1 * dist_to_prey
        components['distance_reward'] = distance_reward
        
        # Formation reward - encourage surrounding the prey
        other_adversaries = [i for i in adversary_indices if i != agent_idx]
        if len(other_adversaries) > 0:
            other_positions = agent_positions[other_adversaries]
            # Calculate angles from prey to all pursuers
            vectors_to_prey = other_positions - prey_position
            angles = []
            for vec in vectors_to_prey:
                angle = np.arctan2(vec[1], vec[0])
                angles.append(angle)
            
            # Check if current agent fills a gap in the formation
            min_gap = np.pi / 3  # 60 degrees threshold for good coverage
            my_vector = agent_positions[agent_idx] - prey_position
            my_angle = np.arctan2(my_vector[1], my_vector[0])
            
            gap_filled = True
            for other_angle in angles:
                angle_diff = abs((my_angle - other_angle + np.pi) % (2 * np.pi) - np.pi)
                if angle_diff < min_gap:
                    gap_filled = False
                    break
            
            formation_reward = 5.0 if gap_filled else -1.0
            components['formation_reward'] = formation_reward
        else:
            components['formation_reward'] = 0.0
        
        # Collision penalty between pursuers
        collision_penalty = 0.0
        for other_idx in other_adversaries:
            dist = inter_agent_distances[agent_idx, other_idx]
            if dist < 0.1:  # Collision threshold
                collision_penalty -= 2.0
        components['collision_penalty'] = collision_penalty
        
        # Boundary penalty
        agent_pos = agent_positions[agent_idx]
        boundary_penalty = 0.0
        if (abs(agent_pos[0]) > world_size * 0.9 or 
            abs(agent_pos[1]) > world_size * 0.9):
            boundary_penalty -= 1.0
        components['boundary_penalty'] = boundary_penalty
        
        # Energy penalty (optional)
        speed = np.linalg.norm(agent_velocities[agent_idx])
        energy_penalty = -0.01 * speed
        components['energy_penalty'] = energy_penalty
        
        # Success bonus when prey is captured
        is_prey_captured = np.any(distances_to_prey <= capture_threshold)
        if is_prey_captured:
            success_bonus = 10.0
            components['success_bonus'] = success_bonus
        else:
            components['success_bonus'] = 0.0
            
    else:
        # Prey reward components
        agent_idx_in_preys = prey_indices.index(agent_idx)
        dist_to_nearest_pursuer = np.min(distances_to_prey)
        
        # Distance reward (positive since we want to maximize distance)
        distance_reward = 0.1 * dist_to_nearest_pursuer
        components['distance_reward'] = distance_reward
        
        # Evasion reward - penalize being surrounded
        pursuer_positions = agent_positions[adversary_indices]
        vectors_from_prey = pursuer_positions - prey_position
        avg_direction = np.mean(vectors_from_prey, axis=0)
        
        # Move away from the average direction of pursuers
        desired_velocity = avg_direction * 0.1
        dot_product = np.dot(prey_velocity, desired_velocity)
        evasion_reward = max(0, dot_product) * 2.0
        components['evasion_reward'] = evasion_reward
        
        # Capture penalty
        capture_penalty = 0.0
        if dist_to_nearest_pursuer <= capture_threshold:
            capture_penalty -= 20.0
        components['capture_penalty'] = capture_penalty
        
        # Boundary penalty
        agent_pos = agent_positions[agent_idx]
        boundary_penalty = 0.0
        if (abs(agent_pos[0]) > world_size * 0.9 or 
            abs(agent_pos[1]) > world_size * 0.9):
            boundary_penalty -= 1.0
        components['boundary_penalty'] = boundary_penalty
        
        # Energy penalty (optional)
        speed = np.linalg.norm(agent_velocities[agent_idx])
        energy_penalty = -0.01 * speed
        components['energy_penalty'] = energy_penalty
    
    total_reward = sum(components.values())
    return total_reward, components
